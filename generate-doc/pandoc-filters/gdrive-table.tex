\b{macro table_line(line, header=False, safe=False)-}
\b{for value in line}
\v{'\textbf{' if header else ''}\v{value if safe else value|texesc}\v{'}' if header else ''}\b{if not loop.last} & \b{else} \\\v{'\midrule\endhead' if header else ''}
\b{endif}
\b{endfor}
\b{- endmacro}
