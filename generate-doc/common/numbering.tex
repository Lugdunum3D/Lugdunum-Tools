% Numbering
\usepackage{titlesec}
\setcounter{secnumdepth}{3}
\makeatletter
\def\@seccntformat#1{\csname #1ignore\expandafter\endcsname\csname the#1\endcsname\quad}
\let\sectionignore\@gobbletwo
\let\latex@numberline\numberline
\def\numberline#1{\if\relax#1\relax\else\latex@numberline{#1}\fi}
\makeatother

\newlength{\titleindent}
\setlength{\titleindent}{1em}

\titleformat
{\chapter} % command
[display] % shape
{\bfseries\LARGE\itshape} % format
{Part. \ \thechapter} % label
{0.5ex} % sep
{
\fontsize{40}{30}\selectfont
} % before-code
[
] % after-code

% Title/Heading Formatting

\titleformat{\section}[hang]{\fontsize{18}{20}\selectfont\bfseries}{\thesection}{1em}{}
\renewcommand\thesection{\Roman{section}.}
\titlespacing*{\section}{1\titleindent}{2ex}{0.1em}

\titleformat{\subsection}[hang]{\fontsize{14}{16}\selectfont\bfseries}{\thesubsection}{1em}{}
\renewcommand\thesubsection{\arabic{subsection}.}
\titlespacing*{\subsection}{2\titleindent}{2ex}{0.1em}

\titleformat{\subsubsection}[hang]{\fontsize{12}{14}\selectfont\bfseries}{\thesubsubsection}{1em}{}
\renewcommand\thesubsubsection{\thesubsection\alph{subsubsection}.}
\titlespacing*{\subsubsection}{3\titleindent}{2ex}{0.1em}

\titleformat{\paragraph}[hang]{\bfseries}{\theparagraph}{1em}{}
\titlespacing*{\paragraph}{4\titleindent}{3.25ex plus 1ex minus .2ex}{0.5em}

\titleformat{\subparagraph}[hang]{\bfseries}{\thesubparagraph}{1em}{}
\titlespacing*{\subparagraph}{5\titleindent}{3.25ex plus 1ex minus .2ex}{0.5em}
